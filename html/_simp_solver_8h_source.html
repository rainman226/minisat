200inline void SimpSolver::thaw(){
201 for (int i = 0; i < frozen_vars.size(); i++){
202 Var v = frozen_vars[i];
203 frozen[v] = 0;
204 if (use_simplification)
205 updateElimHeap(v);
206 }
207 frozen_vars.clear(); }
